JQUERY OBJECT OPACITY

点击我以及将鼠标从我身上移开看效果。

//对于有一堆相同属性的元素
$('.obj-opacity').objOpacity({
    warp:'#warp',//如果不想使用事件代理,就不要使用这个参数,设置false
    hover:'effect',
    event:'mouseover',
    bevent:'mouseout',
    focus:1,
    blur:.1,
    speed:600
});
//对于只有一个属性的元素
$('.single').objOpacity({
    warp:false,
    hover:'effect',
    event:'click',
    bevent:'mouseout',
    focus:1,
    blur:.1,
    speed:600
});
